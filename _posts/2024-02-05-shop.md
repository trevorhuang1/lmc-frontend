<script>
    // Function to purchase an ingredient
    const purchaseIngredient = async (ingredient, points) => {
        try {
            const response = await fetch('http://127.0.0.1:8086/purchase-ingredient', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({ ingredient, points })
            });
            if (response.ok) {
                const data = await response.json();
                alert(`You have purchased ${ingredient} for ${points} points`);
            } else {
                console.error('Failed to purchase ingredient');
                alert('Failed to purchase ingredient');
            }
        } catch (error) {
            console.error('Error:', error);
            alert('An error occurred while purchasing the ingredient');
        }
    };
    // Function to retrieve purchased ingredients
    const getPurchasedIngredients = async () => {
        try {
            const response = await fetch('http://127.0.0.1:8086/purchased-ingredients');
            if (response.ok) {
                const data = await response.json();
                console.log(data);
            } else {
                console.error('Failed to retrieve purchased ingredients');
            }
        } catch (error) {
            console.error('Error:', error);
        }
    };
</script>
