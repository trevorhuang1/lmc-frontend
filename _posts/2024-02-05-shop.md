<script>
    // Function to purchase an ingredient
    const purchaseIngredient = async (ingredient, points) => {
        try {
            const response = await fetch('http://127.0.0.1:8086/purchase-ingredient', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({ ingredient, points })
            });
            if (response.ok) {
                const data = await response.json();
                alert(`You have purchased ${ingredient} for ${points} points`);
                getPurchasedIngredients(); // Update owned ingredients after purchase
            } else {
                console.error('Failed to purchase ingredient');
                alert('Failed to purchase ingredient');
            }
        } catch (error) {
            console.error('Error:', error);
            alert('An error occurred while purchasing the ingredient');
        }
    };
    // Function to retrieve purchased ingredients
    const getPurchasedIngredients = async () => {
        try {
            const response = await fetch('http://127.0.0.1:8086/purchased-ingredients');
            if (response.ok) {
                const data = await response.json();
                const ownedIngredientsContainer = document.getElementById('owned-ingredients');
                data.forEach(({ ingredient, points, quantity }) => {
                    // Check if the ingredient already exists in the table
                    const existingRow = ownedIngredientsContainer.querySelector(`tr[data-ingredient="${ingredient}"]`);
                    if (existingRow) {
                        // If the ingredient exists, update the quantity
                        const quantityCell = existingRow.querySelector('.quantity');
                        const newQuantity = parseInt(quantity) + 1; // Increase the quantity by 1
                        quantityCell.textContent = newQuantity;
                    } else {
                        // If the ingredient doesn't exist, add a new row
                        const row = ownedIngredientsContainer.insertRow();
                        row.setAttribute('data-ingredient', ingredient);
                        const cell1 = row.insertCell(0);
                        const cell2 = row.insertCell(1);
                        const cell3 = row.insertCell(2);
                        cell1.textContent = ingredient;
                        cell1.className = 'ingredient';
                        cell2.textContent = 0; // Set the initial quantity to 0
                        cell2.className = 'quantity';
                        cell3.textContent = points;
                        cell3.className = 'points';
                    }
                });
            } else {
                console.error('Failed to retrieve purchased ingredients');
            }
        } catch (error) {
            console.error('Error:', error);
        }
    };
    // Fetch owned ingredients when the page loads
    getPurchasedIngredients();
</script>
