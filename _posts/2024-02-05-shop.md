    let totalPoints = 0;

    function purchaseIngredient(ingredient, price) {
        if (totalPoints - price >= 0) {
            totalPoints -= price;
            alert(`You purchased ${ingredient} for ${price} points. Total points: ${totalPoints}`);
        } else {
            alert("Insufficient points to purchase this ingredient.");
        }
